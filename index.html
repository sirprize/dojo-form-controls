Button

Representation of a native <button> element. Same API as dijit/form/Button.

Programmatic

require(["dijitive/Button"], function (Button) {
    var button1 = new Button({
        type: 'submit',
        'class': 'btn btn-primary',
        label: '<i class="icon-music icon-white"></i> La La Laaaa',
        onClick: function (e) {
            console.info('clicked?');
            return false;
        }
    }, 'button1');

    button1.startup();
});

Declarative

<script>
    var buttonOnClick = function (ev) {
        console.info('Clicked');
    }
</script>

<button class="btn btn-success" data-dojo-type="dijitive/Button" data-dojo-props="onClick: buttonOnClick">
    <i class="icon-music icon-white"></i> La La Laaaa
</button>

<script>
    require(["dijitive/Button", "dojo/parser"]);
</script>

Checkbox

Representation of a native <input type="checkbox"/> element. Same API as dijit/form/CheckBox.

Programmatic

require(["dijitive/Checkbox"], function (Checkbox) {
    var checkbox1 = new Checkbox({
        name: 'menu',
        value: 'clam chowder',
        checked: true,
        onChange: function (e) {
            // value is false when unchecked
            console.info('value is: ' + this.get('value'));
        }
    }, 'checkbox1');
    
    checkbox1.startup();
});

Declarative

<script>
    var checkboxOnChange = function (ev) {
        console.info('New value is: ' + this.get('value'));
    }
</script>

<input type="checkbox" name="extraCheese" value="on" data-dojo-type="dijitive/Checkbox" data-dojo-props="onChange: checkboxOnChange"/>

<script>
    require(["dijitive/Checkbox", "dojo/parser"]);
</script>

Select

Representation of a native <select> element. Same API as dijit/form/Select.

Programmatic

require(["dijitive/Select"], function (Select) {
    var options = [
        { value: '', label: 'Please select' },
        { value: "steak-and-lobster", label: "Steak & Lobster" },
        { value: "clam-chowder", label: "Clam chowder" },
        { value: "pizza", label: "Pizza", disabled: true },
        { value: "pasta", label: "Pasta" }
    ];
    
    var select1 = new Select({
        name: 'meal1',
        value: 'pasta',
        options: options,
        required: true,
        onChange: function (value) {
            console.info('select.onChange(): ' + value)
        }
    }, 'select1');
    
    select1.watch('value', function (prop, oldVal, val) {
        console.info('select.watch("value")', prop, oldVal, val);
    });
    
    select1.watch('message', function (prop, oldVal, val) {
        console[(val) ? 'error' : 'info']('select.watch("message"):', val);
    });
    
    select1.startup();
});

Textarea

Representation of a native <textarea> element. Same API as dijit/form/SimpleTextarea.

Programmatic

require(["dijitive/Textarea"], function (Textarea) {
    var textarea1 = new Textarea({
        name: 'note',
        autocomplete: false,
        uppercase: true,
        trim: true,
        onKeyPress: function (ev) {
            console.info('key pressed')
        }
    }, 'textarea1');

    textarea1.startup();
    
    textarea1.watch('value', function (name, oldVal, val) {
        console.info('New value is "' + val);
    });
});

Declarative

<script>
    var textareaOnKeyPress = function (ev) {
        console.info('Key pressed')
    }
    
    var textareaOnChange = function (value) {
        console.info('select.onChange(): ' + value)
    }
</script>

<textarea name="note" data-dojo-type="dijitive/Textarea" data-dojo-props="onKeyPress: textareaOnKeyPress, onChange: textareaOnChange, propercase: true"></textarea>

<script>
    require(["dijitive/Textarea", "dojo/parser"]);
</script>
