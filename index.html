Button

Representation of a native <button> element. Same API as dijit/form/Button.

Declarative

<script>
    var buttonOnClick = function (ev) {
        console.info('Clicked');
    }
</script>

<button class="btn btn-success" data-dojo-type="dijitive/Button" data-dojo-props="onClick: buttonOnClick">
    <i class="icon-music icon-white"></i> Click Click
</button>

Programmatic

var b = new Button({
    type: 'submit',
    'class': 'btn btn-primary',
    label: '<i class="icon-music icon-white"></i> Click Click',
    onClick: function (e) {
        console.info('Clicked?');
        return false;
    }
}, 'my-button');

b.startup();

Select

Representation of a native <select> element. Same API as dijit/form/Select.

Declarative

<select name="meal3" data-dojo-type="dijitive/Select" required="true">
    <option value="">Please select</option>
    <option value="steak-and-lobster">Steak &amp; Lobster</option>
    <option value="clam-chowder" selected="selected">Clam Chowder</option>
    <option value="pizza" disabled="true">Pizza</option>
    <option value="pasta">Pasta</option>
    <script type="dojo/aspect" data-dojo-method="onChange" data-dojo-args="value">
        console.info('select.onChange(): ' + value);
    </script>
    
    <script type="dojo/watch" data-dojo-prop="value" data-dojo-args="prop, oldVal, val">
        console.info('select.watch("value")', prop, oldVal, val);
    </script>
    
    <script type="dojo/watch" data-dojo-prop="message" data-dojo-args="prop, oldVal, val">
        console[(val) ? 'error' : 'info']('select.watch("message"):', val);
    </script>
</select>

Programmatic

var options = [
    { value: '', label: 'Please select' },
    { value: "steak-and-lobster", label: "Steak & Lobster" },
    { value: "clam-chowder", label: "Clam Chowder" },
    { value: "pizza", label: "Pizza", disabled: true },
    { value: "pasta", label: "Pasta" }
];

var s = new Select({
    name: 'meal1',
    value: 'pasta',
    options: options,
    required: true,
    onChange: function (value) {
        console.info('select.onChange(): ' + value)
    }
}, 's');

s.watch('value', function (prop, oldVal, val) {
    console.info('select.watch("value")', prop, oldVal, val);
});

s.watch('message', function (prop, oldVal, val) {
    console[(val) ? 'error' : 'info']('select.watch("message"):', val);
});

s.startup();

Textarea

Representation of a native <textarea> element. Same API as dijit/form/SimpleTextarea.

Declarative

<script>
    var textareaOnKeyPress = function (ev) {
        console.info('Key pressed')
    }
    
    var textareaOnChange = function (value) {
        console.info('select.onChange(): ' + value)
    }
</script>

<textarea name="note" data-dojo-type="dijitive/Textarea" data-dojo-props="onKeyPress: textareaOnKeyPress, onChange: textareaOnChange, propercase: true"></textarea>

Programmatic

var ta = new Textarea({
    name: 'note',
    autocomplete: false,
    uppercase: true,
    trim: true,
    onKeyPress: function (ev) {
        console.info('Key pressed')
    }
}, 'ta');

ta.startup();

ta.watch('value', function (name, oldVal, val) {
    console.info('New value is "' + val);
});
